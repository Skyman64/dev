== vs ===

  • == does implicit type conversion when comparing
  • === matches only when the values are equal as our their types
if      (0 == '')   { console.log("0 == ''");  }
else if (0 != '')   { console.log("0 != ''");  }

if      (0 === '')  { console.log("0 === ''"); }
else if (0 !== '')  { console.log("0 !== ''"); }
