Match a Real

^\s*([-]?(\d+|\d+\.\d+|\.\d+|\d+\.))\s*$

import re

v = ['1', '-5', '0', '.1', '-.7', '1.2', '-3.4', '-10.', '91.']
x = ['  a', 'b', 'c']

for _v in v:
  m = re.search('^\s*([-]?(\d+|\d+\.\d+|\.\d+|\d+\.))\s*$', _v)
  if m:
    print "matched:", m.group(1)

for _x in x:
  m = re.search('^\s*([-]?(\d+|\d+\.\d+|\.\d+|\d+\.))\s*$', _x)
  if m:
    print "matched??", m.group(1)
